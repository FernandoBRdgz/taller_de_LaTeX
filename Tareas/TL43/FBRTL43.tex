\section*{Parámetros de \texttt{/includegraphics} }

\bigskip
\bigskip
\begin{enumerate}[a.]
\item \textbf{width:}\\ Indica a \LaTeX\ la Anchura de la imágen. (debe especificarse preferentemente en centímetros)
\item \textbf{height:}\\ Indica a \LaTeX\ la Altura de la imágen. (debe especificarse preferentemente en centímetros)
\item \textbf{textwidth:}\\ Indica a \LaTeX\ que debe Escalar la imágen de forma ``ajustada".
\item \textbf{keepaspectratio:}\\ Indica a \LaTeX\ que Mantenga el aspecto en forma de cuadrado independientemente de si se indica una altura distinta a la anchura.
\item \textbf{scale:}\\ Indica a \LaTeX\ la Escala de la imágen. (debe especificarse preferentemente en porcentaje)
\item \textbf{draft:}\\ Indica a \LaTeX\ que ``imprima" la Ruta donde se ubica guardada la imágen.
\item \textbf{angle:}\\ Indica a \LaTeX\ el Ángulo de la imágen. (debe especificarse preferentemente en grados)
\end{enumerate}
